    // Función para buscar APKs
    function searchAPK() {
        let input = document.getElementById('search').value.toLowerCase();
        let apkCards = document.querySelectorAll('.apk-card');

        apkCards.forEach(card => {
            let apkName = card.getAttribute('data-name').toLowerCase();
            if (apkName.includes(input)) {
                card.style.display = "block";
            } else {
                card.style.display = "none";
            }
        });
    }

    // Función para filtrar por categoría
    function filterAPKs(category) {
        let apkCards = document.querySelectorAll('.apk-card');

        apkCards.forEach(card => {
            let apkCategory = card.getAttribute('data-category');
            if (category === 'all' || apkCategory === category) {
                card.style.display = "block";
            } else {
                card.style.display = "none";
            }
        });
    }

    // Función para mostrar créditos
    function showCredits() {
        let creditsText = document.getElementById('credits-text');
        creditsText.style.display = "block";
    }
